<?php
header('Content-Type: application/json');

// Database connection details
$servername = "localhost";
$username = "root";
$password = "";
$dbname = "leave_management_db";

// Create connection
$conn = new mysqli($servername, $username, $password, $dbname);

// Check connection
if ($conn->connect_error) {
    die(json_encode(['success' => false, 'message' => 'Connection failed: ' . $conn->connect_error]));
}

// Get JSON input and decode it
$input = json_decode(file_get_contents('php://input'), true);

try {
    // Update the leave request status in the 'leave_applications' table to 'Approved'
    $usernamePrefix = '4AL21CS';
    $sql_update_leave = "UPDATE leave_applications SET status = 'Approved' WHERE username LIKE ?";
    $stmt_update = $conn->prepare($sql_update_leave);
    $param = $usernamePrefix . "%";
    $stmt_update->bind_param("s", $param);

    if ($stmt_update->execute()) {
        // Fetch details of all approved leave applications
        $sql_select_approved = "SELECT username, start_date, end_date, reason, NOW() as approved_at FROM leave_applications WHERE username LIKE ? AND status = 'Approved'";
        $stmt_select = $conn->prepare($sql_select_approved);
        $stmt_select->bind_param("s", $param);
        $stmt_select->execute();
        $result = $stmt_select->get_result();

        // Store details in approved_leave_applications table
        $sql_insert_approved = "INSERT INTO approved_leave_applications (username, start_date, end_date, reason, approved_at) VALUES (?, ?, ?, ?, ?)";
        $stmt_insert = $conn->prepare($sql_insert_approved);

        while ($row = $result->fetch_assoc()) {
            $stmt_insert->bind_param("sssss", $row['username'], $row['start_date'], $row['end_date'], $row['reason'], $row['approved_at']);
            $stmt_insert->execute();
        }

        // Check if rows were inserted successfully
        if ($stmt_insert->affected_rows > 0) {
            echo json_encode(['success' => true, 'message' => 'Leave applications marked as approved and stored successfully']);
        } else {
            echo json_encode(['success' => false, 'message' => 'No rows inserted into approved_leave_applications table']);
        }
    } else {
        echo json_encode(['success' => false, 'message' => 'Error updating leave status: ' . $stmt_update->error]);
    }
} catch (Exception $e) {
    echo json_encode(['success' => false, 'message' => 'Exception: ' . $e->getMessage()]);
}

// Close statements and connection
$stmt_update->close();
$stmt_select->close();
$stmt_insert->close();
$conn->close();
?>
