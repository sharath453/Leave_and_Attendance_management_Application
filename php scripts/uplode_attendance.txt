<?php
require 'db_connection.php'; // Make sure this file connects to your database

$data = json_decode(file_get_contents('php://input'), true);

if (isset($data['student_id'], $data['subject'], $data['attendance_percentage'])) {
    $studentId = $data['student_id'];
    $subject = $data['subject'];
    $attendancePercentage = $data['attendance_percentage'];

    $sql = "INSERT INTO attendance (student_id, subject, attendance_percentage) VALUES (?, ?, ?)
            ON DUPLICATE KEY UPDATE attendance_percentage = ?";
    $stmt = $conn->prepare($sql);
    $stmt->bind_param("isdi", $studentId, $subject, $attendancePercentage, $attendancePercentage);

    if ($stmt->execute()) {
        echo json_encode(array("success" => true));
    } else {
        echo json_encode(array("error" => $stmt->error));
    }
} else {
    echo json_encode(array("error" => "Invalid input"));
}
?>
